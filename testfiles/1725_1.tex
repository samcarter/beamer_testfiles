\documentclass{beamer}

\usepackage{xpatch}

\makeatletter
\newif\ifbeamer@hide
\define@key{beamerframe}{hide}[true]{\beamer@hidetrue}
\BeforeBeginEnvironment{frame}{\beamer@hidefalse}
\xpatchcmd{\beamer@@@@frame}{%
  \gdef\beamer@whichframes{#1}%
}{
  \ifbeamer@hide
    \gdef\beamer@whichframes{beamer:0}%
  \else
    \gdef\beamer@whichframes{#1}%
  \fi
}{}{}
\makeatother
