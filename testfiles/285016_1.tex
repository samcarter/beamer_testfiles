    \end{beamercolorbox}
    \begin{beamercolorbox}[wd=\paperwidth,ht=2.125ex,dp=1.125ex,leftskip=.3cm,rightskip=.3cm plus1fil]{section in head/foot}
        \usebeamerfont{subsubsection in head/foot}%
