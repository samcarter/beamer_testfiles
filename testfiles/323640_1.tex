\documentclass{beamer}

\usetheme{Berlin}

\usepackage{tikz}

\AtBeginSection[]
{
    \begingroup
    \setbeamercolor{background dark}{bg=backgrounddark}
    \setbeamertemplate{headline}{}
    \begin{frame}
        \vspace{2em}\usebeamerfont{section title}
        \vspace*{-\headheight}
        \begin{tikzpicture}
            \node[anchor=west, blue, inner sep=0pt] at (0,0) {\insertsection};
        \end{tikzpicture}
    \end{frame}
    \endgroup
}
