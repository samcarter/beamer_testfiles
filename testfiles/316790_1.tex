    \vfill
    \centering
    \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{section page}
        \usebeamerfont{title}%
        \textit{Chap. \thesection~}%
                {\color{black} \insertsectionhead}\par%
    \end{beamercolorbox}
    \vfill
    \end{frame}
}
