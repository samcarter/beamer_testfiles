\documentclass[11pt]{beamer}

\usepackage[listings]{tcolorbox}

\begin{document}
\begin{frame}[fragile]
  \begin{tcblisting}{colback=red!5!white,colframe=red!75!black}
    long plus(long x, long y);

    void sumstore(long x, long y, long *dest)
    {
      long t = plus(x, y);
      *dest = t;
    }
  \end{tcblisting}
  \pause
  \begin{tcblisting}{colback=red!5!white,colframe=red!75!black}
    sumstore:
      pushq   %rbx
      movq    %rdx, %rbx
      call    plus
      movq    %rax, (%rbx)
      popq    %rbx
      ret
  \end{tcblisting}
\end{frame}


\end{document}
