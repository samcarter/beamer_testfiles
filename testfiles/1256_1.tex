    \usebeamerfont{section title}

    \ifx\insertsubsectionhead\@empty\else%

      \usebeamerfont{subsection title}%
      \insertsubsectionhead
    \fi
  \end{minipage}
  \par
  \vspace{\baselineskip}
}
