\documentclass{beamer}

\newcommand{\floattitle}[1]{%
    \begin{beamercolorbox}[sep=0.3cm,left,wd=\paperwidth]{frametitle}
     \usebeamerfont{frametitle}%
     \vbox{}\vskip-1ex%
     \strut#1\strut\par%
     \vskip-1ex%
    \end{beamercolorbox}%
}

\end{frame}
\end{document}
