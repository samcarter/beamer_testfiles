\documentclass[t]{beamer}

\def\beamer@typesetheadorfoot#1{% typeset the head or footline
    {\parskip0pt %\offinterlineskip%
        \hbox{%
            \hskip-\Gm@lmargin%
            \hbox{%
                \vbox{%
                    \@tempdima=\textwidth%
                    \textwidth=\paperwidth%
                    \hsize=\textwidth%
                    \def\\{,}%
                    \usebeamertemplate***{#1}%
                }}%
                \hskip-\Gm@rmargin%
            }}\hfil%
        }

        }%
    \end{minipage}%
