\setbeamerfont{note page}{size=\tiny}

\addtobeamertemplate{note page}{\setbeamerfont{itemize/enumerate subbody}{size=\tiny}}{}

      \item This
      \end{itemize}
    \end{itemize}
  \note{This is small}
  \note[item]{This is small
  \begin{itemize}
    \item This is large
  \end{itemize}}
\end{frame}

\end{document}
