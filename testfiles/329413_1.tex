    \only<1>{%
        \addtocontents{toc}{%
         \protect\beamer@subsectionintoc{\the\c@section}{0}{%
                    \insertframenumber\quad%
              \ifx\insertframesubtitle\@empty%
                  \insertframetitle%
              \else%
                  \insertframesubtitle%
              \fi%
         }{\the\c@page}{\the\c@part}{\the\beamer@tocsectionnumber}%
        }%
    }%
}%
