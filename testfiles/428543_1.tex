\begin{frame}
\frametitle{Limit of a Discontinuous Function}\pause
Let us find \[\lim_{x \to 0} f(x) = \left\{
\begin{array}{ll}
          (2x+3) & x \leq 0 \\
          3(x+1) & x > 0 \\
          \end{array}
          \right. \]
and \[\lim_{x \to 1} f(x) = \left\{
\begin{array}{ll}
          (2x+3) & x \leq 0 \\
          3(x+1) & x > 0 \\
          \end{array}
          \right. \]
\end{frame}
